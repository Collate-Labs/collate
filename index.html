<!DOCTYPE html>

<html>

  <head>

    <title>FlightOS Chat</title>

  </head>

  <body>

    <h1>Welcome to FlightOS Chat</h1>


    <div id="chat-box" style="border:1px solid gray; height:200px; overflow:auto; padding:10px; margin-bottom:10px;">

      <!-- Messages will show up here -->

    </div>


    <input type="text" id="chat-input" placeholder="Type your message here..." style="width: 70%;" />

    <button onclick="sendMessage()">Send</button>


    <script>

      const chatBox = document.getElementById("chat-box");

      const input = document.getElementById("chat-input");


      function sendMessage() {

        const msg = input.value.trim();

        if (msg !== "") {

          const p = document.createElement("p");

          p.textContent = "You: " + msg;

          chatBox.appendChild(p);

          input.value = "";

          chatBox.scrollTop = chatBox.scrollHeight;

        }

      }


      input.addEventListener("keypress", function (e) {

        if (e.key === "Enter") {

          sendMessage();

        }

      });

    </script>

  </body>

</html>
